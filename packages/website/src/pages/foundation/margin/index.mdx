Usage

### Margin

Utilities for controlling an element's margin.

<Table>
  <Table.Head>
    <Table.Row>
      <Table.Cell>ClassName</Table.Cell>
      <Table.Cell>Value (px)</Table.Cell>
      <Table.Cell>Value (rem)</Table.Cell>
      <Table.Cell>Example</Table.Cell>
    </Table.Row>
  </Table.Head>
  <Table.Body>
    {Object.keys(tokens).map(token => {
      const value = tokens[token]
      const valuePx = 1 * parseFloat(value) * 16
      const clearValue = token.replace('spacing-', '')
      const margins = ['m', 'my', 'mx', 'mb', 'mt', 'ml', 'mr']
      return margins.map(m => (
        <Table.Row key={`${m}-${token}`}>
          <Table.Cell>
            <strong>{`${m}-${clearValue}`}</strong>
          </Table.Cell>
          <Table.Cell>
            <Tag>{valuePx}px</Tag>
          </Table.Cell>
          <Table.Cell>
            <Tag>{value}</Tag>
          </Table.Cell>
          <Table.Cell>
            <div
              style={{
                width: value,
                height: 20,
                backgroundColor: '#0052CC',
              }}
            />
          </Table.Cell>
        </Table.Row>
      ))
    })}
  </Table.Body>
</Table>
